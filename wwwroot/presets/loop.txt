Counter <- 1
WHILE Counter <= 3 DO

    FOR i <- 1 TO 3

        AnotherCounter <- 1
        REPEAT
            PRINT Counter & ", " & i & ", " & AnotherCounter
            AnotherCounter <- AnotherCounter + 1
        UNTIL AnotherCounter > 3

    NEXT i

    Counter <- Counter + 1
ENDWHILE


FOR i <- 1 TO 10
    IF i = 6 THEN
        PRINT "BREAK"
        BREAK
    ELSE IF i = 3 THEN
        PRINT "CONTINUE"
        CONTINUE
    ENDIF
    PRINT i
NEXT i
